import sendNotification from '../middleware/notification';
import { Message, User, Group } from '../models';
 
/**
 * This class handles messages
 * @class MessageClass
 */
class MessageClass {
 
  /**
   * This method handles posting of messages in a group
   * @static
   *
   * @param {object} request
   * @param {object} response
   *
   * @returns {object} Promise
   *
   * @memberof MessageClass
   */
  static create(request, response) {
    if (!request.body.message || !request.body.priority) {
      return response
        .status(400)
        .json('All fields are requestuired');
    }
    const { userId, email, username } = request.decoded;

  }
 
  /**
   * This method handles retrieving messages in a group
   * @static
   *
   * @param {object} request
   * @param {object} response
   *
   * @returns {object} Promise
   *
   * @memberof MessageClass
   */
  static retrieve(request, response) {
    Message.findAll({
      where: {
        GroupId: request.params.groupId
      },
      include: [
        {
          model: User,
          attributes: [
            'id',
            'username',
            'fullname',
            'email'
          ]
        }
      ]
    }).then((messageRetrieved) => {
      response.status(200).json(messageRetrieved);
    }).catch(error => response.status(500).json(error.responseponse.data));
  }
}
export default MessageClass;
