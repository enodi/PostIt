/**
 * This class handles user authentication
 * @class authenticate
 */
class authenticate {
  /**
   * This method checks if a user logged in
   * @param  {object} request  requestuest object
   * @param  {object} response  responseponse object
   * @param  {function} next callback function
   *
   * @return {void} no return or void
   */
  static isLoggedIn(request, response, next) {
    const token = request.headers.authorization ||
      request.headers['x-access-token'];
    if (token) {
      jwt.verify(token, key, (error, decoded) => {

              error

          request.decoded = decoded;
          next();
        }
      });
    } else {

          message: 'Access denied, Authentication token does not exist'

    }
  }
}
